import { useState, useEffect, useCallback } from 'react';
import {
  MapPin,
  Thermometer,
  Droplets,
  Clock,
  Activity,
  ChevronLeft,
  RefreshCw,
  Wind,
  ShieldCheck,
  AlertTriangle,
  ActivityIcon
} from 'lucide-react';
import { useNavigate, useParams } from 'react-router-dom';
import api from '../api/Sapi';

interface Sensor {
  id: number;
  name: string;
  serial_number: string;
  description: string;
  location: string;
  location_lat: number;
  location_lng: number;
  status: string;
  created_at: string;
}
interface Gateway {
  id: number;
  name: string;
  serial_number: string;
  description: string;
  location: string;
  location_lat: number;
  location_lng: number;
  status: string;
  created_at: string;
}

interface SensorTelemetry {
  id: number;
  temperature: number;
  humidity: number;
  time_taken: string;
}

interface GatewayResponse {
  gateway: Gateway;
  sensors: Sensor[];
}

const Sensors: React.FC = () => {
  // --- State ---
  const [sensors, setSensors] = useState<Sensor[]>([]);
  const [gateway, setGateway] = useState<Gateway | null>(null);

  const [selectedSensor, setSelectedSensor] = useState<Sensor | null>(null);
  const [sensorData, setSensorData] = useState<SensorTelemetry[]>([]);
  const { id } = useParams<{ id: string }>();
  const navigate = useNavigate();

  const gatewayId = Number(id);

  const [loading, setLoading] = useState({
    gateways: false,
    sensors: false,
    data: false,
    refreshing: false
  });

  const onBack = () => {
    navigate(-1); // go back
  };
  // --- Fetching Logic ---
  const fetchSensors = useCallback(async () => {
    try {
      setLoading(prev => ({ ...prev, sensors: true }));
      // Fetching sensors filtered by gatewayId if provided
      const endpoint = gatewayId
        ? `/gateways/${gatewayId}/data`
        : `/sensors`;

      const { data } = await api.get<GatewayResponse>(endpoint);

      // ✅ sensors (array)
      setSensors(
        data.sensors.map((s: Sensor) => ({
          ...s,
          location_lat: Number(s.location_lat),
          location_lng: Number(s.location_lng),
        }))
      );

      // ✅ gateway (single object)
      setGateway({
        ...data.gateway,
        location_lat: Number(data.gateway.location_lat),
        location_lng: Number(data.gateway.location_lng),
      });

    } catch (error) {
      console.error('Error fetching sensors:', error);
    } finally {
      setLoading(prev => ({ ...prev, sensors: false }));
    }
  }, [gatewayId]);

  const fetchSensorData = useCallback(async (sensorId: string, isSilent = false) => {
    if (!isSilent) setLoading(prev => ({ ...prev, data: true }));
    else setLoading(prev => ({ ...prev, refreshing: true }));

    try {
      const { data } = await api.get(`/fruitfly/${sensorId}/data`);
      setSensorData(data);
    } catch (error) {
      console.error('Error fetching sensor data:', error);
    } finally {
      setLoading(prev => ({ ...prev, data: false, refreshing: false }));
    }
  }, []);

  // Initial Load
  useEffect(() => {
    fetchSensors();
  }, [fetchSensors]);

  // Real-time Update Loop for Selected Sensor
  useEffect(() => {
    if (!selectedSensor) return;

    fetchSensorData(selectedSensor.serial_number);

    const interval = setInterval(() => {
      fetchSensorData(selectedSensor.serial_number, true);
    }, 60000 * 10);

    return () => clearInterval(interval);
  }, [selectedSensor, fetchSensorData]);

  // --- Helpers ---
  const getStatusColor = (status?: string): string => {

    switch (status?.toLowerCase()) {
      case 'active': return 'bg-emerald-100 text-emerald-700 border-emerald-200';
      case 'warning': return 'bg-amber-100 text-amber-700 border-amber-200';
      default: return 'bg-slate-100 text-slate-600 border-slate-200';
    }
  };

  return (
    <div className="min-h-screen bg-slate-50 p-4 md:p-8">
      <div className="max-w-7xl mx-auto">

        {/* Breadcrumbs / Back Navigation */}
        <div className="mb-6 flex items-center justify-between">
          <div className="flex items-center gap-4">
            <button
              onClick={onBack}
              className="p-2 hover:bg-white rounded-full border border-transparent hover:border-slate-200 transition-all text-slate-500"
            >
              <ChevronLeft size={24} />
            </button>

            <div>
              {loading.gateways ? (
                <div className="text-slate-400">Loading gateway...</div>
              ) : !gateway ? (
                <div className="text-slate-400 italic">Gateway not found.</div>
              ) : (
                <>
                  <h1 className="text-2xl font-bold text-slate-900">
                    Sensor Network
                  </h1>
                  <p className="text-slate-500 text-sm">
                    Telemetrics for Gateway <b>{gateway.name}</b>
                  </p>
                  <p className="text-slate-500 text-sm">
                    Gateway ID: {gateway.serial_number}
                  </p>
                </>
              )}
            </div>
          </div>


          <div className="flex items-center gap-3">
            {loading.refreshing && (
              <span className="flex items-center gap-2 text-xs font-medium text-indigo-500 animate-pulse">
                <RefreshCw size={14} className="animate-spin" /> Syncing Live Data
              </span>
            )}
            <button
              onClick={fetchSensors}
              className="p-2 text-slate-400 hover:text-indigo-600 bg-white border border-slate-200 rounded-lg shadow-sm"
            >
              <RefreshCw size={18} />
            </button>
          </div>
        </div>

        <div className="grid grid-cols-1 lg:grid-cols-12 gap-8">

          {/* Left Column: Sensors List */}
          <div className="lg:col-span-4 space-y-4">
            <div className="bg-white rounded-2xl shadow-sm border border-slate-200 overflow-hidden">
              <div className="p-4 border-b border-slate-100 bg-slate-50/50">
                <h2 className="font-bold text-slate-800 flex items-center gap-2">
                  <Activity size={18} className="text-indigo-500" />
                  Connected Nodes ({sensors.length})
                </h2>
              </div>

              <div className="divide-y divide-slate-100 max-h-[calc(100vh-300px)] overflow-y-auto">
                {loading.sensors ? (
                  <div className="p-12 text-center text-slate-400">Loading sensors...</div>
                ) : sensors.length === 0 ? (
                  <div className="p-12 text-center text-slate-400 italic">No sensors found in this region.</div>
                ) : (
                  sensors.map((sensor) => (
                    <div
                      key={sensor.id}
                      onClick={() => setSelectedSensor(sensor)}
                      className={`p-4 cursor-pointer transition-all hover:bg-slate-50 relative group ${selectedSensor?.id === sensor.id ? 'bg-indigo-50/50' : ''
                        }`}
                    >
                      {selectedSensor?.id === sensor.id && (
                        <div className="absolute right-0 top-0 bottom-0 w-1 bg-indigo-600" />
                      )}
                      <div className="flex justify-between items-start">
                        <div className="flex gap-3">
                          <div className={`p-2.5 rounded-xl ${selectedSensor?.id === sensor.id ? 'bg-indigo-600 text-white shadow-lg shadow-indigo-100' : 'bg-slate-100 text-slate-400'
                            }`}>
                            <Wind size={20} />
                          </div>
                          <div>
                            <h3 className="font-bold text-slate-800 leading-tight">{sensor.name}</h3>
                            <div className="flex items-center gap-1 text-xs text-slate-500 mt-1">
                              <MapPin size={12} /> {sensor.location}. 
                              <div className='text-blue-700  text-xs flex items-center  gap-1'>
                              <ActivityIcon size={12} />ID: {sensor.serial_number}
                              </div>
                            </div>
                          </div>
                        </div>
                        <span className={`px-2 py-0.5 text-[10px] font-bold uppercase tracking-wider rounded border ${getStatusColor(sensor.status)}`}>
                          {sensor.status}
                        </span>
                      </div>
                    </div>
                  ))
                )}
              </div>
            </div>
          </div>

          {/* Right Column: Sensor Details & Data */}
          <div className="lg:col-span-8">
            {selectedSensor ? (
              <div className="space-y-6 animate-in fade-in duration-300">

                {/* Header Stats */}
                <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
                  <div className="bg-white p-6 rounded-2xl shadow-sm border border-slate-200 flex items-center gap-6">
                    <div className="w-16 h-16 bg-red-50 text-red-500 rounded-2xl flex items-center justify-center">
                      <Thermometer size={32} />
                    </div>
                    <div>
                      <p className="text-slate-500 text-sm font-medium">Temperature</p>
                      <h4 className="text-3xl font-black text-slate-900">
                        {sensorData[0]?.temperature ?? '--'}<span className="text-slate-400 text-xl">°C</span>
                      </h4>
                    </div>
                  </div>

                  <div className="bg-white p-6 rounded-2xl shadow-sm border border-slate-200 flex items-center gap-6">
                    <div className="w-16 h-16 bg-blue-50 text-blue-500 rounded-2xl flex items-center justify-center">
                      <Droplets size={32} />
                    </div>
                    <div>
                      <p className="text-slate-500 text-sm font-medium">Humidity</p>
                      <h4 className="text-3xl font-black text-slate-900">
                        {sensorData[0]?.humidity ?? '--'}<span className="text-slate-400 text-xl">%</span>
                      </h4>
                    </div>
                  </div>
                </div>

                {/* Main Data View */}
                <div className="bg-white rounded-2xl shadow-sm border border-slate-200 overflow-hidden">
                  <div className="p-6 border-b border-slate-100 flex items-center justify-between">
                    <div>
                      <h3 className="font-bold text-slate-900">Historical Logs</h3>
                      <p className="text-xs text-slate-400 flex items-center gap-1 mt-1">
                        {/* <Clock size={12} /> Last reading: {sensorData[0] ? new Date(sensorData[0].time_taken).toLocaleString() : 'N/A'} */}
                        <Clock size={12} /> Last reading: {sensorData.length > 0
                          ? new Date(
                            sensorData.reduce((latest, current) =>
                              new Date(current.time_taken) > new Date(latest.time_taken) ? current : latest
                            ).time_taken
                          ).toLocaleString()
                          : 'N/A'}
                      </p>
                    </div>
                    <div className="flex gap-2">
                      <span className="flex items-center gap-1.5 px-3 py-1 bg-emerald-50 text-emerald-600 text-xs font-bold rounded-full border border-emerald-100">
                        <ShieldCheck size={14} /> Encrypted
                      </span>
                    </div>
                  </div>

                  {loading.data ? (
                    <div className="p-20 text-center text-slate-400 flex flex-col items-center gap-3">
                      <RefreshCw className="animate-spin text-indigo-500" size={32} />
                      <p className="font-medium">Pulling latest telemetrics...</p>
                    </div>
                  ) : sensorData.length === 0 ? (
                    <div className="p-20 text-center">
                      <div className="inline-flex p-4 rounded-full bg-slate-50 text-slate-300 mb-4">
                        <AlertTriangle size={32} />
                      </div>
                      <p className="text-slate-500 font-medium">No telemetry data recorded for this sensor.</p>
                    </div>
                  ) : (
                    <div className="overflow-x-auto">
                      <table className="w-full text-left border-collapse">
                        <thead>
                          <tr className="bg-slate-50/50">
                            <th className="px-6 py-4 text-xs font-bold text-slate-500 uppercase tracking-wider">Timestamp</th>
                            <th className="px-6 py-4 text-xs font-bold text-slate-500 uppercase tracking-wider">Temp</th>
                            <th className="px-6 py-4 text-xs font-bold text-slate-500 uppercase tracking-wider">Humidity</th>
                            <th className="px-6 py-4 text-xs font-bold text-slate-500 uppercase tracking-wider">Delta</th>
                          </tr>
                        </thead>
                        <tbody className="divide-y divide-slate-100">
                          {sensorData.slice(0, 5).map((log, idx) => {
                            const prev = sensorData[idx + 1];
                            const currTemp = Number(log.temperature);
                            const prevTemp = prev ? Number(prev.temperature) : currTemp;
                            const diff = Number((currTemp - prevTemp).toFixed(1));

                            return (
                              <tr key={log.id} className="hover:bg-slate-50/80 transition-colors">
                                <td className="px-6 py-4 whitespace-nowrap">
                                  <div className="flex flex-col">
                                    <span className="text-sm font-medium text-slate-700">
                                      {new Date(log.time_taken).toLocaleTimeString()}
                                    </span>
                                    <span className="text-[10px] text-slate-400">
                                      {new Date(log.time_taken).toLocaleDateString()}
                                    </span>
                                  </div>
                                </td>
                                <td className="px-6 py-4 whitespace-nowrap">
                                  <span className="text-sm font-bold text-slate-900">{log.temperature}°C</span>
                                </td>
                                <td className="px-6 py-4 whitespace-nowrap">
                                  <span className="text-sm font-bold text-slate-900">{log.humidity}%</span>
                                </td>
                                <td className="px-6 py-4 whitespace-nowrap">
                                  <span className={`text-xs font-bold px-2 py-0.5 rounded-full ${diff > 0 ? 'text-orange-600 bg-orange-50' :
                                    diff < 0 ? 'text-blue-600 bg-blue-50' :
                                      'text-slate-400 bg-slate-100'
                                    }`}>
                                    {diff > 0 ? '+' : ''}{diff}°
                                  </span>
                                </td>
                              </tr>
                            );
                          })}
                        </tbody>
                      </table>
                    </div>
                  )}
                </div>
              </div>
            ) : (
              <div className="h-full min-h-[500px] flex flex-col items-center justify-center bg-white rounded-2xl border-2 border-dashed border-slate-200 p-12 text-center">
                <div className="w-24 h-24 bg-slate-50 rounded-full flex items-center justify-center text-slate-300 mb-6">
                  <Wind size={48} />
                </div>
                <h3 className="text-xl font-bold text-slate-900">No Sensor Active</h3>
                <p className="text-slate-500 mt-2 max-w-sm">
                  Select a specific sensor node from the list to view live telemetry, environmental trends, and historical logs.
                </p>
              </div>
            )}
          </div>
        </div>
      </div>
    </div>
  );
};

export default Sensors;
